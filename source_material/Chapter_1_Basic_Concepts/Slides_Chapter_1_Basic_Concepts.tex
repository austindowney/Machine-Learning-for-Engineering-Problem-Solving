\section{1.2~Definition of Machine Learning}


\begin{frame}{Definition of Machine Learning}
\begin{itemize}
\item Systems improve with experience
\item Learn from data, not explicit rules
\end{itemize}
\vspace{0.5em}
\begin{itemize}
\item \textbf{Arthur Samuel (1959):} ``Field of study that gives computers the ability to learn without being explicitly programmed''
\item \textbf{Tom Mitchell (1997):} ``A computer program is said to learn from experience $E$ with respect to some task $T$ and some performance measure $P$, if its performance on $T$, as measured by $P$, improves with experience  $E$.''
\end{itemize}
\end{frame}

%------------------------------------------------

\begin{frame}{Learning as a Function}
\centering

\begin{equation*}
Y = f(X)
\end{equation*}
\vspace{0.3em}
\begin{equation*}
Y = f(X) + e
\end{equation*}
\vspace{0.3em}
\begin{itemize}
\item $X$: input features
\item $Y$: output (target)
\item $f$: unknown function
\item $e$: error (noise)
\end{itemize}
\vspace{0.3em}
\begin{itemize}
\item Training: build the model
\item Inference: use the model
\end{itemize}
\end{frame}

\section{1.3~Supervision in Machine Learning}

\begin{frame}{Supervised Learning}
\textbf{Key idea:} Data comes with labels \\
\vspace{1em}
Learn a mapping from inputs $\rightarrow$ outputs
\vspace{1.5em} \\
\textbf{Two main tasks:}
\begin{itemize}
\item Classification (categories)
\item Regression (numeric values)
\end{itemize}
\vspace{1em} 
\textit{Goal: generalize to new, unseen data}
\end{frame}

%------------------------------------------------

\begin{frame}{Supervised Learning in Practice}

Training uses labeled data, where each example includes input features and a known target.

\vspace{1em}

\textbf{Examples:}
\begin{itemize}
\item Spam filter (spam vs ham)
\item House price prediction
\end{itemize}

\vspace{1em}

\textbf{Common algorithms:}
\begin{itemize}
\item k-Nearest Neighbors
\item Linear Regression
\item Logistic Regression
\item Support Vector Machines (SVMs)
\item Decision Trees / Random Forests
\end{itemize}

\end{frame}




%------------------------------------------------

\begin{frame}{Unsupervised Learning}

Unsupervised learning uses unlabeled data to discover structure and patterns automatically.

\vspace{1em}

\textbf{Key tasks:}
\begin{itemize}
\item \textbf{Clustering}: group similar data
\item \textbf{Association}: discover relationships
\item \textbf{Dimension reduction}: simplify data
\end{itemize}

\vspace{1em}

\textbf{Common algorithms:}
\begin{itemize}
\item k-Means
\item Hierarchical Clustering
\item Expectation Maximization
\end{itemize}

\end{frame}

\begin{frame}{Unsupervised Learning: Concepts and Methods}

Unsupervised learning reveals structure in data without labels.

\vspace{1em}

\textbf{Key concepts:}
\begin{itemize}
\item \textbf{Clustering}: group similar data points
\item \textbf{Association}: discover relationships between variables
\item \textbf{Dimension reduction}: reduce number of variables
\end{itemize}

\vspace{1em}

\textbf{Common algorithms:}
\begin{itemize}
\item k-Means
\item Hierarchical Clustering (HCA)
\item Expectation Maximization
\end{itemize}

\end{frame}

\begin{frame}{Semi-Supervised Learning}

Semi-supervised learning uses a small amount of labeled data with a large amount of unlabeled data.

\vspace{1em}

\begin{itemize}
\item Combines supervised and unsupervised learning
\item Leverages structure in unlabeled data
\end{itemize}

\vspace{1em}

\textbf{Example:}
\begin{itemize}
\item Group similar images automatically (clustering)
\item Label a few examples
\item Propagate labels to remaining data
\end{itemize}

\end{frame}

\section{1.4~Types of Learning (Batch and Online)}
\begin{frame}{Types of Learning}
\begin{itemize}
\item How models learn from data
\item Two main approaches:
\begin{itemize}
\item Batch learning (offline)
\item Online learning (incremental)
\end{itemize}
\item Trade-off: accuracy vs adaptability vs resources
\end{itemize}
\end{frame}

%------------------------------------------------

\begin{frame}{Batch Learning}
\centering
\includegraphics[width=0.8\textwidth]{../figures/training_batch}
\vspace{0.3em}
\small \\
Batch learning framework with separate training and inference stages.
\vspace{0.5em}

\begin{itemize}
\item Train once on full dataset
\item No learning after deployment
\item Requires retraining from scratch to update
\item Computationally expensive
\end{itemize}
\end{frame}

%------------------------------------------------

\begin{frame}{Online Learning}
\centering
\includegraphics[width=0.8\textwidth]{../figures/training_online}
\vspace{0.3em}
\small \\
Online learning framework where data is used to continuously training (or update / fine-tune) the model.
\vspace{0.5em}

\begin{itemize}
\item Learns incrementally from new data
\item Adapts continuously
\item Works for streaming or large datasets
\item Learning rate controls adaptation speed
\end{itemize}
\end{frame}

\begin{frame}{Bad Data in Machine Learning}
\centering
\includegraphics[width=0.6\textwidth]{../figures/cows}
\vspace{0.3em}
\small \\ 
Training a machine learning algorithm to recognize cows may end up just learning to recognize grass. humorously called ``short-cut learning''\protect\footnotemark[1].

\section{1.5~Learning Approaches: Instance-Based and Model-Based}
\begin{frame}{Instance-Based Learning}
\centering
\includegraphics[width=0.7\textwidth]{../figures/instance_based_learning}
\vspace{0.3em}
\small \\
Instance-based learning where the class of a unknown instance is inferred from the its distance to data points with known labels.
\vspace{0.5em}

\begin{itemize}
\item Memorizes training examples
\item Uses similarity (distance) to make predictions
\item Flexible but can be slow
\end{itemize}
\end{frame}

%------------------------------------------------

\begin{frame}{Model-Based Learning}
\centering
\includegraphics[width=0.7\textwidth]{../figures/model_based_learning}
\vspace{0.3em}
\small \\
Model-based learning where the class of a unknown instance is inferred from its location in reference to a model trained on the data with known labels.
\vspace{0.5em}

\begin{itemize}
\item Learns a model from data
\item Uses model to make predictions
\item Faster at inference, depends on model quality
\end{itemize}
\end{frame}
